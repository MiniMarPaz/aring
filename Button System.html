Design System · Components

Button

Neutral, brand-agnostic button system. Four hierarchies × three sizes × six states, driven entirely by --btn-* custom properties so the system is themeable without touching component code.

Usage · React / TSX

// Install: pnpm add @ds/button
import { Button } from "@ds/button";
import "@ds/button/dist/index.css";

export function Example() {
  return (
    <div>
      <Button hierarchy="primary">Save</Button>
      <Button hierarchy="secondary" loading>Syncing…</Button>
      <Button hierarchy="tertiary" leadingIcon={<Plus />}>Add</Button>
      <Button hierarchy="ghost" iconOnly aria-label="Close"
              leadingIcon={<X />} />
      <Button hierarchy="primary" danger>Delete</Button>
    </div>
  );
}
